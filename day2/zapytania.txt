SELECT * FROM `clients` WHERE `gender` = "male" AND `date_of_birth` > "1999-12-31"

SELECT `gender`, COUNT(*) AS 'ilosc' FROM `clients` GROUP BY `gender`

SELECT `id`,`name`,`surname`, `orders_count` FROM `clients`
ORDER BY `orders_count` DESC LIMIT 1

SELECT * FROM `clients` WHERE `name` LIKE 'T%' OR `surname` LIKE 'T%'

SELECT COUNT(DISTINCT`city`) AS `ilosc` FROM `clients`
